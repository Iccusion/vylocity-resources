textStyle: a JSON object containing styles for the text (ex: {"padding": 0, "fill": "#000", "leading": 0} )
{
fontFamily: default font to be used for text displays
fontSize: default size of font in pixels
fontStyle: style of font; 'normal', 'italic', or 'oblique'
fontVariant: variant of font; 'normal' or 'small-caps'
fontWeight: weight of font; 'normal', 'bold', 'bolder', 'lighter', '100', '200', '300', '400', '500', '600', '700', '800', or '900' textBaseline: where vertically to start drawing the characters on the line; top/alphabetic/hanging/middle/ideographic/bottom
fill: default color of text for this element; can be an array of colors
fillGradientType: a number; fill must be an array for gradients; 0 for vertical gradient, 1 for horizontal
fillGradientStops: if fillGradientType is an array, this can be an array of numbers 0 to 1 to change gradient spacing
stroke: style of text outline for this element; currently supports colors
strokeThickness: width of text outline
padding: padding between the edge of the element and the text
leading: default spacing between each line of text
letterSpacing: default 0; space between each letter
lineHeight: height of line in pixels
lineJoin: default 'miter'; type of corner used; 'miter', 'round', or 'bevel'
miterLimit: default 10; miter limit to use when using miter linJoin
align: default 'left'; possible values 'left', 'right', 'center', 'start', 'end';
breakWords: determines if lines can wrap between letters of a word
dropShadow: enable a drop shadow
dropShadowAlpha: alpha of the drop shadow
dropShadowAngle: angle of the drop shadow in radians
dropShadowBlur: number representing drop shadow blur radius
dropShadowColor: fill style to be used for drop shadow
dropShadowDistance: default 5; number representing distance of drop shadow
trim: boolean true or false to trim transparent borders
wordWrap: default false; boolean true or false to determine if lines should be broken after a certain width
wordWrapWidth: default 100; number of pixels before word wrap
anchor: default {'x': 0, 'y': 0}; object containing x and y anchor; ex {'x': 0.5, 'y': 0.5}
scale: default {'x': 1, 'y': 1}; object containing x and y scale; ex {'x': 2, 'y': 2}
}
